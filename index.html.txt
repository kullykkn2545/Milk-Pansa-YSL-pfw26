function generateCaption(){
  if(used.length === captions.length){
    used = [];
  }

  let available = captions.filter(c => !used.includes(c));
  let random = available[Math.floor(Math.random()*available.length)];

  document.getElementById("captionBox").style.opacity = 0;

    document.getElementById("captionBox").innerText = current;

  used.push(random);
  updateCounter();
}

function copyCaption(){
